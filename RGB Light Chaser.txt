/*  
RGB Light Chaser Program

Program Goal: Lights up RGB colors in a sequence, then reverses from the last color back to the first.

Theme: Digital Output
*/

int timeDelay = 10; // Variable to set LED delay in milliseconds

// setup() runs once when the board powers on or is reset
void setup() {
    // Set RGB pins as outputs
    pinMode(10, OUTPUT);
    pinMode(11, OUTPUT);
    pinMode(12, OUTPUT);
}

// loop() runs continuously in a never-ending cycle
void loop() {
    // Turn on all three LEDs
    digitalWrite(10, 1);
    digitalWrite(11, 1);
    digitalWrite(12, 1);
    delay(timeDelay); // Keep LEDs on for 10 ms

    // Turn on only pin 10
    digitalWrite(10, 1);
    digitalWrite(11, 0);
    digitalWrite(12, 0);
    delay(timeDelay);

    // Turn on only pin 11
    digitalWrite(10, 0);
    digitalWrite(11, 1);
    digitalWrite(12, 0);
    delay(timeDelay);

    // Turn on only pin 12
    digitalWrite(10, 0);
    digitalWrite(11, 0);
    digitalWrite(12, 1);
    delay(timeDelay);
}
