/*  
RGB Colour Selector Program

Program Goal: Uses PWM pins to adjust the intensity of each RGB color, allowing the creation of colors other than pure Red, Green, or Blue.

Theme: Analog Output
*/

int red = 11;   // Connect the Red color of the RGB to pin 11
int green = 10; // Connect the Green color of the RGB to pin 10
int blue = 9;   // Connect the Blue color of the RGB to pin 9

// setup() runs once when the board powers on or is reset
void setup() {
    // Set PWM pins as outputs
    pinMode(11, OUTPUT); // Red
    pinMode(10, OUTPUT); // Green
    pinMode(9, OUTPUT);  // Blue
}

// loop() runs continuously in a never-ending cycle
void loop() {
    // Set analog values for each color to mix and create a final color
    analogWrite(red, 250);   // Adjust Red intensity
    analogWrite(green, 200); // Adjust Green intensity
    analogWrite(blue, 2);    // Adjust Blue intensity
}
