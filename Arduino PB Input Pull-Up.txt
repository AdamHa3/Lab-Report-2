/*
Arduino Push Button (PB) Input Pull-Up

Program Goal: Detects the state of a push button (PB) using an internal pull-up resistor. 
When pressed, the button reads 0; when not pressed, it reads 1, similar to a standard Arduino pull-up but with an internal resistor.

Theme: Digital Input
*/

int pushButton = 7; // Assign pin 7 to the push button

// setup() runs once when the board powers on or is reset
void setup() {
    pinMode(pushButton, INPUT);  // Set PB pin as input
    Serial.begin(9600);          // Start Serial Monitor communication
}

// loop() runs continuously in a never-ending cycle
void loop() {
    int buttonState = digitalRead(pushButton); // Read the push button state

    // Display button state on the Serial Monitor
    Serial.print("buttonState = "); 
    Serial.println(buttonState);

    delay(1); // Small delay between Serial Monitor prints
}
