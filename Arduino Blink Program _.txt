/*
Blinking LED

Program Goal: Turns an LED on for 1 second, then off for 2 seconds, repeatedly.

Theme: Digital Output
*/

// setup() runs once when the board powers on or is reset
void setup() {
    pinMode(13, OUTPUT); // Set pin 13 as an output
}

// loop() runs continuously in a never-ending cycle
void loop() {
    digitalWrite(13, HIGH); // Turn LED on
    delay(1000);             // Keep LED on for 1 second

    digitalWrite(13, LOW);  // Turn LED off
    delay(2000);             // Keep LED off for 2 seconds
}
