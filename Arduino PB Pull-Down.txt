/*
Arduino Push Button (PB) Pull-Down

Program Goal: The push button (PB) reads 0 when not pressed and 1 when pressed.

Theme: Digital Input
*/

int pushButton = 7; // Assign pin 7 of the Arduino to the push button (PB)

// setup() runs once when the board powers on or is reset
void setup() {
    pinMode(pushButton, INPUT);  // Set PB pin as input
    Serial.begin(9600);          // Establish Serial Monitor communication
}

// loop() runs continuously in a never-ending cycle
void loop() {
    int buttonState = digitalRead(pushButton); // Read the push button state

    // Display button state on the Serial Monitor
    Serial.print("buttonState = "); 
    Serial.println(buttonState);

    delay(1); // Small delay between Serial Monitor prints
}
